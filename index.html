Quickstart

npm install @actionpackd/sdk-core

Basic Example

import { serverCompose } from '@actionpackd/sdk-core';

const schema = {
  type: 'object',
  properties: {
    summary: { type: 'string' },
    sentiment: { type: 'string', enum: ['positive', 'neutral', 'negative'] }
  }
};

const response = await serverCompose(
  "Analyze this tweet: 'Just launched our new AI SDK!'",
  schema
);
